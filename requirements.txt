
# HRMS Frontend - Setup Documentation
# -----------------------------------

# This file lists the NPM packages required to run the frontend application.
# It is structured for usage with Node.js and NPM.

# LOCATION
# --------
# You must run installation commands inside the 'front_end' directory:
# Path: ./front_end/

# INSTALLATION COMMANDS
# ---------------------
# To install all dependencies at once, run:
# cd front_end
# npm install


# DEPENDENCIES LIST
# -----------------

# Core React Framework
react@^19.2.0
react-dom@^19.2.0

# Routing
react-router-dom@^7.11.0  # Handles navigation (Login, Admin, Employee pages)

# UI & Styling
tailwindcss@^4.1.18      # Utility-first CSS framework
@tailwindcss/postcss@^4.1.18 # PostCSS plugin for Tailwind v4
lucide-react@^0.562.0    # Icon library

# Build Tools & Configuration (Dev Dependencies)
vite@^7.2.4              # Fast frontend build tool
postcss@^8.5.6           # CSS Tooling
autoprefixer@^10.4.23    # CSS vendor prefixing

# Linting (Dev Dependencies)
eslint@^9.39.1
@vitejs/plugin-react@^5.1.1

# ----------

#    cd front_end
#
